12: int   NbMetals = ...;
13: int   NbRaw = ...;
14: int   NbScrap = ...;
15: int   NbIngo = ...;
16: 
17: range Metals = 1..NbMetals;
18: range Raws = 1..NbRaw;
19: range Scraps = 1..NbScrap;
20: range Ingos = 1..NbIngo;
21: 
22: float CostMetal[Metals] = ...;
23: float CostRaw[Raws] = ...;
24: float CostScrap[Scraps] = ...;
25: float CostIngo[Ingos] = ...;
26: float Low[Metals] = ...;
27: float Up[Metals] = ...;
28: float PercRaw[Metals][Raws] = ...;
29: float PercScrap[Metals][Scraps] = ...;
30: float PercIngo[Metals][Ingos] = ...;
31: 
32: int Alloy  = ...;
33: dvar float+    p[Metals];
34: dvar float+    r[Raws];
35: dvar float+    s[Scraps];
36: dvar int+      i[Ingos];
37: dvar float    m[j in Metals] in Low[j] * Alloy .. Up[j] * Alloy;
38: 
39: 
40: minimize 
41:   sum(j in Metals) CostMetal[j] * p[j] +
42:   sum(j in Raws)   CostRaw[j]   * r[j] +
43:   sum(j in Scraps) CostScrap[j] * s[j] +
44:   sum(j in Ingos)  CostIngo[j]  * i[j];
45: subject to {
46:   forall( j in Metals )
47:     ct1:
48:       m[j] == 
49:       p[j] + 
50:       sum( k in Raws )   PercRaw[j][k] * r[k] +
51:       sum( k in Scraps ) PercScrap[j][k] * s[k] +
52:       sum( k in Ingos )  PercIngo[j][k] * i[k];
53:     ct2:  
54:       sum( j in Metals ) m[j] == Alloy;
55: }
