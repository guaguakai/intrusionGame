12: /* ------------------------------------------------------------
13: 
14: Problem Description
15: -------------------
16: 
17: The problem involves choosing colors for the countries on a map in 
18: such a way that at most four colors (blue, white, yellow, green) are 
19: used and no neighboring countries are the same color. In this exercise, 
20: you will find a solution for a map coloring problem with six countries: 
21: Belgium, Denmark, France, Germany, Luxembourg, and the Netherlands. 
22: 
23: ------------------------------------------------------------ */
24: using CP;
25: 
26: range r = 0..3;
27: 
28: execute{
29:         }
30: 
31: 
32: string Names[r] = ["blue", "white", "yellow", "green"]; 
33: 
34: dvar int Belgium in r;
35: dvar int Denmark in r;
36: dvar int France in r;
37: dvar int Germany  in r;
38: dvar int Luxembourg in r;
39: dvar int Netherlands in r;
40: 
41: subject to {  
42:    Belgium != France; 
43:    Belgium != Germany; 
44:    Belgium != Netherlands;
45:    Belgium != Luxembourg;
46:    Denmark != Germany; 
47:    France != Germany; 
48:    France != Luxembourg; 
49:    Germany != Luxembourg;
50:    Germany != Netherlands;    
51: }
52: 
53: execute {
54:    writeln("Belgium:     ", Names[Belgium]);
55:    writeln("Denmark:     ", Names[Denmark]);
56:    writeln("France:      ", Names[France]);
57:    writeln("Germany:     ", Names[Germany]);
58:    writeln("Luxembourg:  ", Names[Luxembourg]);
59:    writeln("Netherlands: ", Names[Netherlands]);
60: }
